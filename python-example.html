# 1. Import Libraries
import pandas as pd
import numpy as np
from datetime import datetime

# 2. Define a function to process and clean data
def analyze_sales_data(file_path):
    """Loads, cleans, and analyzes a simple sales dataset."""
    print("--- Starting Data Analysis ---")

    # Load the dataset
    try:
        df = pd.read_csv(file_path)
    except FileNotFoundError:
        print(f"Error: File not found at {file_path}")
        return

    # Basic Cleaning: Drop rows with any missing values
    df_cleaned = df.dropna()
    print(f"Original Rows: {len(df)}. Cleaned Rows: {len(df_cleaned)}")

    # Data Transformation: Ensure 'Date' is a datetime object
    df_cleaned['Date'] = pd.to_datetime(df_cleaned['Date'], errors='coerce')

    # Feature Engineering: Calculate Profit
    if 'Revenue' in df_cleaned.columns and 'Cost' in df_cleaned.columns:
        df_cleaned['Profit'] = df_cleaned['Revenue'] - df_cleaned['Cost']
    else:
        print("Warning: 'Revenue' or 'Cost' columns missing. Cannot calculate profit.")
        return

    # Aggregate Analysis: Calculate key metrics
    total_profit = df_cleaned['Profit'].sum()
    avg_revenue = df_cleaned['Revenue'].mean()
    
    print("\n--- Key Metrics ---")
    print(f"Total Profit: ${total_profit:,.2f}")
    print(f"Average Revenue per Transaction: ${avg_revenue:,.2f}")
    
    # Show the first 5 rows of the processed data
    print("\n--- Head of Processed Data ---")
    print(df_cleaned.head().to_string())
